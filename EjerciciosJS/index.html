1. Invertir una Cadena

Descripción: Escribe una función que tome una cadena y devuelva la cadena invertida.

            function invertirCadena(cadena) {
                return cadena.split('').reverse().join('');
            }

            // Ejemplo de uso
            console.log(invertirCadena('Hola Mundo')); // 'odnuM aloH'

2. Número Primo

Descripción: Escribe una función que determine si un número es primo.

                function esNumeroPrimo(numero) {
                    if (numero <= 1) return false;
                    for (let i = 2; i < numero; i++) {
                        if (numero % i === 0) return false;
                    }
                    return true;
                }

                // Ejemplo de uso
                console.log(esNumeroPrimo(7)); // true
                console.log(esNumeroPrimo(10)); // false

3. Eliminar Duplicados en un Arreglo

Descripción: Escribe una función que elimine los elementos duplicados de un arreglo.

            function eliminarDuplicados(arreglo) {
                return [...new Set(arreglo)];
            }

            // Ejemplo de uso
            console.log(eliminarDuplicados([1, 2, 2, 3, 4, 4, 5])); // [1, 2, 3, 4, 5]

4. Factorial de un Número

Descripción: Escribe una función que calcule el factorial de un número.

            function factorial(numero) {
                if (numero < 0) return -1;
                if (numero === 0) return 1;
                return numero * factorial(numero - 1);
            }

            // Ejemplo de uso
            console.log(factorial(5)); // 120

5. Contar Vocales

Descripción: Escribe una función que cuente el número de vocales en una cadena.

            function contarVocales(cadena) {
                const vocales = 'aeiouAEIOU';
                let contador = 0;
                for (let letra of cadena) {
                    if (vocales.includes(letra)) {
                        contador++;
                    }
                }
                return contador;
            }

            // Ejemplo de uso
            console.log(contarVocales('Hola Mundo')); // 4

6. Sumar los Números de un Arreglo

Descripción: Escribe una función que sume todos los números de un arreglo.

            function sumarNumeros(arreglo) {
                return arreglo.reduce((acc, num) => acc + num, 0);
            }

            // Ejemplo de uso
            console.log(sumarNumeros([1, 2, 3, 4, 5])); // 15

7. Buscar el Máximo en un Arreglo

Descripción: Escribe una función que encuentre el número máximo en un arreglo.

            function encontrarMaximo(arreglo) {
                return Math.max(...arreglo);
            }

            // Ejemplo de uso
            console.log(encontrarMaximo([1, 2, 3, 4, 5])); // 5
